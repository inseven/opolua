PROC xborder:(type%, flags%)
	LOCAL x%
	gXBORDER type%, flags%, w%, h%
	x% = gX
	gMOVE 7, h%-10
	gPRINT type%, HEX$(flags%)
	gAT x%, gY
	gMOVE 5 + w%, -(h%-10)
	checkEnd:	
ENDP

PROC border:(flags%)
	LOCAL x%
	gBORDER flags%, w%, h%
	x% = gX
	gMOVE 5, h%-8
	gPRINT HEX$(flags%)
	gAT x%, gY
	gMOVE 5 + w%, -(h%-8)
	checkEnd:	
ENDP

PROC newline:
	gAT 8, gY + h% + 8
ENDP

PROC checkEnd:
	IF gX + w% > gWIDTH
		newline:
	ENDIF
ENDP

PROC but:(type%, state%)
	gBUTTON HEX$(type%) + "b" + HEX$(state%), type%, w%, h%, state%
	gMOVE 5 + w%, 0
	checkEnd:
ENDP
